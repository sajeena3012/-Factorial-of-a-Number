import java.util.Scanner; public class Solution { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); int N = scanner.nextInt(); scanner.close(); int evenSum = 0; int oddSum = 0; while (N > 0) { int digit = N % 10; if (digit % 2 == 0) { evenSum += digit; } else { oddSum += digit; } N /= 10; } System.out.println(evenSum + " " + oddSum); } }
